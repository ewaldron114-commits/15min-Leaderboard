        const STATE = {
            apiKey: null,
            isScanning: false,
            scanTimer: null,
            apiCallCount: 0,
            tickersUniverse: [],
            results: [],
            alertLog: [],
        };
        const $ = (id) => document.getElementById(id);
        const dom = {
            apiKeySection: $('apiKeySection'), apiKeyInput: $('apiKeyInput'),
            saveKeyBtn: $('saveKeyBtn'), clearKeyBtn: $('clearKeyBtn'),
            controls: $('controls'), sessionIndicator: $('sessionIndicator'),
            sessionText: $('sessionText'), marketClock: $('marketClock'),
            startScanBtn: $('startScanBtn'), stopScanBtn: $('stopScanBtn'),
            statusBar: $('statusBar'), scanStatus: $('scanStatus'),
            lastScan: $('lastScan'), apiCalls: $('apiCalls'),
            tickersScanned: $('tickersScanned'), results: $('results'),
            resultsBody: $('resultsBody'), resultCount: $('resultCount'),
            alerts: $('alerts'), alertLog: $('alertLog'),
        };
        const EXCHANGE_CODES = ['XNAS', 'XNYS', 'XASE'];
        function formatMarketCap(v) {
            if (!v || v === 0) return 'N/A';
            if (v >= 1e9) return '$' + (v / 1e9).toFixed(2) + 'B';
            if (v >= 1e6) return '$' + (v / 1e6).toFixed(1) + 'M';
            return '$' + v.toLocaleString();
        }
        function formatVolume(v) {
            if (!v) return 'N/A';
            if (v >= 1e6) return (v / 1e6).toFixed(1) + 'M';
            if (v >= 1e3) return (v / 1e3).toFixed(0) + 'K';
            return v.toString();
        }
        function formatPercent(v) {
            if (v === null || v === undefined || isNaN(v)) return '--';
            return v.toFixed(2) + '%';
        }
        function getCurrentSession() {
            const now = new Date();
            const et = new Date(now.toLocaleString('en-US', { timeZone: 'America/New_York' }));
            const h = et.getHours(), m = et.getMinutes(), d = et.getDay();
            if (d === 0 || d === 6) return 'closed';
            const t = h * 60 + m;
            if (t >= 240 && t < 570) return 'pre';
            if (t >= 570 && t < 960) return 'regular';
            if (t >= 960 && t < 1200) return 'post';
            return 'closed';
        }
        function updateClock() {
            const now = new Date();
            const et = new Date(now.toLocaleString('en-US', { timeZone: 'America/New_York' }));
            dom.marketClock.textContent = 
                `${et.getHours().toString().padStart(2,'0')}:${et.getMinutes().toString().padStart(2,'0')}:${et.getSeconds().toString().padStart(2,'0')} ET`;
        }
        function updateSessionIndicator() {
            const s = getCurrentSession();
            const dot = dom.sessionIndicator.querySelector('.dot');
            dot.classList.remove('active','closed');
            if (s === 'pre') { dom.sessionText.textContent = 'Pre-Market (4am-9:30am)'; dot.classList.add('active'); }
            else if (s === 'regular') { dom.sessionText.textContent = 'Regular Hours (9:30am-4pm)'; dot.classList.add('active'); }
            else if (s === 'post') { dom.sessionText.textContent = 'Post-Market (4pm-8pm)'; dot.classList.add('active'); }
            else { dom.sessionText.textContent = 'Market Closed'; dot.classList.add('closed'); }
        }
        function addAlert(msg, type) {
            const time = new Date().toLocaleTimeString('en-US', { timeZone: 'America/New_York' });
            STATE.alertLog.unshift({ time, message: msg, type });
            if (STATE.alertLog.length > 100) STATE.alertLog.pop();
            renderAlerts();
        }
        function renderAlerts() {
            dom.alertLog.innerHTML = STATE.alertLog.map(a => 
                `<div class="alert-entry"><span class="time">${a.time} ET</span> ${a.message}</div>`).join('');
            dom.alerts.style.display = STATE.alertLog.length ? 'block' : 'none';
        }
        async function polygonRequest(endpoint, params = {}) {
            if (!STATE.apiKey) throw new Error('No API key');
            const url = new URL(`https://api.polygon.io${endpoint}`);
            url.searchParams.append('apiKey', STATE.apiKey);
            Object.entries(params).forEach(([k,v]) => url.searchParams.append(k,v));
            const res = await fetch(url.toString());
            if (res.status === 429) throw new Error('Rate limit exceeded');
            if (!res.ok) throw new Error(`API error: ${res.status}`);
            STATE.apiCallCount++;
            dom.apiCalls.textContent = `API calls: ${STATE.apiCallCount}`;
            return res.json();
        }
        async function fetchAllTickers() {
            const all = [];
            let cursor = null;
            addAlert('Fetching ticker universe...');
            do {
                const p = { market: 'stocks', exchange: EXCHANGE_CODES.join(','), active: 'true', type: 'CS', limit: 1000 };
                if (cursor) p.cursor = cursor;
                const data = await polygonRequest('/v3/reference/tickers', p);
                if (data.results) all.push(...data.results.map(t => t.ticker));
                cursor = data.next_url ? new URL(data.next_url).searchParams.get('cursor') : null;
                await new Promise(r => setTimeout(r, 200));
            } while (cursor);
            addAlert(`Loaded ${all.length} tickers`);
            return all;
        }
        async function fetchSnapshots(tickers) {
            const all = [];
            for (let i = 0; i < tickers.length; i += 250) {
                const batch = tickers.slice(i, i+250).join(',');
                try {
                    const data = await polygonRequest('/v2/snapshot/locale/us/markets/stocks/tickers', { tickers: batch });
                    if (data.tickers) all.push(...data.tickers);
                } catch(e) { console.warn('Batch fail:', e.message); }
                await new Promise(r => setTimeout(r, 300));
            }
            return all;
        }
        async function fetchBars(ticker, timespan='minute', multiplier=15, limit=10) {
            const today = new Date().toISOString().split('T')[0];
            try {
                const data = await polygonRequest(`/v2/aggs/ticker/${ticker}/range/${multiplier}/${timespan}/${today}/${today}`, { adjusted:'true', sort:'desc', limit });
                return data.results || [];
            } catch(e) { return []; }
        }
        function calculateROC(bars, periods) {
            if (!bars || bars.length < periods+1) return null;
            return ((bars[0].c - bars[periods].c) / bars[periods].c) * 100;
        }
        function getFilters() {
            return {
                minMarketCap: parseFloat($('minMarketCap').value)*1e6,
                maxMarketCap: parseFloat($('maxMarketCap').value)*1e6,
                minPrice: parseFloat($('minPrice').value),
                minVolume: parseInt($('minVolume').value),
                maxFloat: parseFloat($('maxFloat').value)*1e6,
                minROC: parseFloat($('minROC').value),
                rocTimeframe: $('rocTimeframe').value,
            };
        }
        function filterBySnapshot(snapshots, f) {
            const q = [];
            for (const item of snapshots) {
                const day = item.day || {}, det = item.details || {};
                const mc = day.market_cap || 0, px = day.c || 0, vol = day.v || 0;
                const flt = det.float_shares_outstanding || null;
                if (mc < f.minMarketCap || mc > f.maxMarketCap) continue;
                if (px < f.minPrice) continue;
                if (vol < f.minVolume) continue;
                if (flt && flt > f.maxFloat) continue;
                const o = day.o || px;
                q.push({ ticker: item.ticker, price: px, marketCap: mc, volume: vol, floatShares: flt, sessionROC: ((px-o)/o)*100, dayOpen: o });
            }
            q.sort((a,b) => Math.abs(b.sessionROC)-Math.abs(a.sessionROC));
            return q.slice(0, 100);
        }
        async function deepScan(candidates, f) {
            const r = [];
            dom.scanStatus.textContent = `Deep scanning ${candidates.length} candidates...`;
            for (let i=0; i<candidates.length; i++) {
                const c = candidates[i];
                try {
                    const bars = await fetchBars(c.ticker);
                    if (bars.length >= 5) {
                        const r15 = calculateROC(bars,1), r1h = calculateROC(bars,4);
                        let maxR = 0;
                        if (f.rocTimeframe==='15min') maxR = r15||0;
                        else if (f.rocTimeframe==='1hour') maxR = r1h||0;
                        else maxR = Math.max(r15||0, r1h||0);
                        if (Math.abs(maxR) >= f.minROC) {
                            r.push({ ...c, roc15min: r15, roc1hour: r1h, maxROC: maxR, floatRotation: c.floatShares ? (c.volume/c.floatShares*100) : null });
                        }
                    }
                } catch(e) {}
                dom.scanStatus.textContent = `Scanning ${i+1}/${candidates.length}...`;
                await new Promise(rr => setTimeout(rr, 300));
            }
            r.sort((a,b) => Math.abs(b.maxROC)-Math.abs(a.maxROC));
            return r.slice(0, 30);
        }
        async function runScan() {
            if (!STATE.apiKey) { addAlert('Enter API key first'); return; }
            if (getCurrentSession()==='closed') { dom.scanStatus.textContent='Market closed - waiting...'; return; }
            const f = getFilters(), start = Date.now();
            try {
                dom.scanStatus.textContent = 'Fetching snapshots...';
                const snaps = await fetchSnapshots(STATE.tickersUniverse);
                dom.tickersScanned.textContent = `Tickers: ${snaps.length}`;
                const candidates = filterBySnapshot(snaps, f);
                dom.scanStatus.textContent = `${candidates.length} candidates after filter`;
                STATE.results = await deepScan(candidates, f);
                renderResults();
                const dur = ((Date.now()-start)/1000).toFixed(1);
                dom.lastScan.textContent = `Last scan: ${new Date().toLocaleTimeString()} (${dur}s)`;
                dom.scanStatus.textContent = `Complete - ${STATE.results.length} movers`;
                if (STATE.results.length) addAlert(`🔥 ${STATE.results.length} movers! Top: ${STATE.results[0].ticker} (${formatPercent(STATE.results[0].maxROC)})`);
            } catch(e) { dom.scanStatus.textContent = `Error: ${e.message}`; addAlert(`Scan failed: ${e.message}`); }
        }
        function renderResults() {
            if (!STATE.results.length) {
                dom.resultsBody.innerHTML = '<tr class="placeholder"><td colspan="9">No movers found matching criteria</td></tr>';
                dom.resultCount.textContent = '0 results'; return;
            }
            dom.resultsBody.innerHTML = STATE.results.map(r => {
                const c15 = r.roc15min>=0?'roc-positive':'roc-negative';
                const c1h = r.roc1hour>=0?'roc-positive':'roc-negative';
                const strong = Math.abs(r.maxROC)>=5 ? ' roc-strong' : '';
                return `<tr>
                    <td><strong>${r.ticker}</strong></td>
                    <td>$${r.price?.toFixed(2)||'--'}</td>
                    <td>${formatMarketCap(r.marketCap)}</td>
                    <td class="${c15}${strong}">${formatPercent(r.roc15min)}</td>
                    <td class="${c1h}${strong}">${formatPercent(r.roc1hour)}</td>
                    <td>${formatVolume(r.volume)}</td>
                    <td>${r.floatShares?formatVolume(r.floatShares):'N/A'}</td>
                    <td>${r.floatRotation?r.floatRotation.toFixed(1)+'%':'N/A'}</td>
                    <td>${getCurrentSession().toUpperCase()}</td>
                </tr>`;
            }).join('');
            dom.resultCount.textContent = `${STATE.results.length} results`;
        }
        function startScanning() {
            if (STATE.isScanning) return;
            STATE.isScanning = true;
            dom.startScanBtn.disabled = true;
            dom.stopScanBtn.disabled = false;
            dom.scanStatus.textContent = 'Starting...';
            runScan();
            STATE.scanTimer = setInterval(runScan, parseInt($('scanInterval').value)*1000);
            addAlert('Scanner started');
        }
        function stopScanning() {
            STATE.isScanning = false;
            dom.startScanBtn.disabled = false;
            dom.stopScanBtn.disabled = true;
            if (STATE.scanTimer) { clearInterval(STATE.scanTimer); STATE.scanTimer = null; }
            dom.scanStatus.textContent = 'Stopped';
            addAlert('Scanner stopped');
        }
        function saveApiKey() {
            const k = dom.apiKeyInput.value.trim();
            if (!k) { addAlert('Enter valid key'); return; }
            STATE.apiKey = k;
            localStorage.setItem('polygon_api_key', k);
            dom.apiKeySection.style.display = 'none';
            dom.controls.style.display = 'flex';
            dom.statusBar.style.display = 'flex';
            dom.results.style.display = 'block';
            dom.alerts.style.display = 'block';
            dom.clearKeyBtn.style.display = 'inline-block';
            addAlert('Key saved. Loading universe...');
            initApp();
        }
        function clearApiKey() {
            STATE.apiKey = null;
            localStorage.removeItem('polygon_api_key');
            dom.apiKeySection.style.display = 'flex';
            dom.controls.style.display = 'none';
            dom.statusBar.style.display = 'none';
            dom.results.style.display = 'none';
            dom.alerts.style.display = 'none';
            dom.clearKeyBtn.style.display = 'none';
            dom.apiKeyInput.value = '';
            stopScanning();
            addAlert('Key cleared');
        }
        async function initApp() {
            if (!STATE.apiKey) return;
            try {
                if (!STATE.tickersUniverse.length) STATE.tickersUniverse = await fetchAllTickers();
                addAlert('Ready. Press Start Scanning.');
                dom.scanStatus.textContent = 'Ready';
            } catch(e) { addAlert(`Init error: ${e.message}`); }
        }
        dom.saveKeyBtn.addEventListener('click', saveApiKey);
        dom.clearKeyBtn.addEventListener('click', clearApiKey);
        dom.startScanBtn.addEventListener('click', startScanning);
        dom.stopScanBtn.addEventListener('click', stopScanning);
        dom.apiKeyInput.addEventListener('keypress', e => { if (e.key==='Enter') saveApiKey(); });
        const savedKey = localStorage.getItem('polygon_api_key');
        if (savedKey) {
            STATE.apiKey = savedKey;
            dom.apiKeySection.style.display = 'none';
            dom.controls.style.display = 'flex';
            dom.statusBar.style.display = 'flex';
            dom.results.style.display = 'block';
            dom.alerts.style.display = 'block';
            dom.clearKeyBtn.style.display = 'inline-block';
            initApp();
        }
        setInterval(updateClock, 1000);
        setInterval(updateSessionIndicator, 30000);
        updateClock();
        updateSessionIndicator();
    </script>
