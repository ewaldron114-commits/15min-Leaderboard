async function fetchFMPUniverse(){
  const urls = {
    gainers: `https://financialmodelingprep.com/stable/biggest-gainers?apikey=${FMP_KEY}`,
    losers:  `https://financialmodelingprep.com/stable/biggest-losers?apikey=${FMP_KEY}`,
    actives: `https://financialmodelingprep.com/stable/most-actives?apikey=${FMP_KEY}`
  };

  console.log('Fetching:', urls[mode]); // for debugging

  const r = await fetch(urls[mode]);
  console.log('FMP Status:', r.status);

  if (!r.ok) {
    throw new Error(`FMP error ${r.status} - Check your API key or regenerate it`);
  }
  
  let list = await r.json();
  if (list && typeof list === 'object' && !Array.isArray(list)) {
    list = list.data || list;
  }

  if (!Array.isArray(list)) throw new Error('Unexpected FMP response');

  const syms = [];
  for (const item of list.slice(0, 20)) {
    const sym = item.symbol || item.ticker;
    if (!sym) continue;
    syms.push(sym);
    data[sym] = {
      symbol: sym,
      name: (item.name || item.companyName || sym).substring(0, 22),
      price: parseFloat(item.price || item.lastPrice || item.close || 0),
      dayChg: parseFloat(item.changesPercentage || item.changePercentage || item.change || 0),
      chg: parseFloat(item.change || 0),
      vol: parseInt(item.volume || 0),
      chg15: data[sym]?.chg15 || 0,
    };
  }
  return syms;
}

  const inp=document.getElementById('symInp');

    const r=await fetch(`https://financialmodelingprep.com/stable/quote/${sym}?apikey=${FMP_KEY}`);
    const d=await r.json();
    if(d && !Array.isArray(d)){ 
      data[sym]={symbol:sym, name:(d.name||sym).substring(0,22), price:d.price||0, dayChg:d.changesPercentage||0, chg:d.change||0, vol:d.volume||0, chg15:0};
    }
